<?php

namespace jwvirtualassistant\AccountManagement;

use GuzzleHttp\Psr7\ServerRequest;
use \PDO;

$BASE_DIR = __DIR__ . "/../../";

require "${BASE_DIR}vendor/autoload.php";

$db = new PDO(
    sprintf("sqlite:%s", DATABASE_FILE)
);
$Request = ServerRequest::fromGlobals();
$query_params = $Request->getQueryParams();

if (!key_exists("account_secret", $query_params)) {
    header("Location: /400Error.html");
}

if (!key_exists("username", $query_params)) {
    header("Location: /400Error.html");
}

$account_secret = $query_params["account_secret"];
$username = $query_params["username"];

if (!usernameAndAccountSecretMatchAnAccount(
    $db,
    new Username($username),
    $account_secret
)) {
    header(
        sprintf(
            "Location: /verify-username?account_secret=%s&error_code=%d",
            htmlspecialchars($account_secret),
            404 // error code
        )
    );
}

if (!accountSecretExists($db, $account_secret)) {
    header(
        sprintf(
            "Location: /verify-account-secret?error_code=%d",
            404 //error code
        )
    );
}
